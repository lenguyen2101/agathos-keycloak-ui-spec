/* ========================================

/* 1. Alert box → first field spacing (screens 2 & 3) */
.alert-warning {
  margin-bottom: 20px;
}

#kc-content-wrapper {
  margin-top: 20px;
}

/* 2. Consistent field group gaps (screens 2 & 3) */
.pf-c-form__group {
  margin-bottom: 16px;
}

/* 3. Helper text — lighter weight (screen 3) */
.pf-c-form__helper-text,
.form-group .help-block {
  font-weight: 400;
  font-size: 12px;
  color: #888;
  margin-top: 4px;
  margin-bottom: 0;
}

/* 4. Card padding — more breathing room (screens 1 & 4) */
.card-pf {
  padding: 32px 40px;
}

/* 5. Body text readability (screens 1 & 4) */
#kc-content-wrapper p {
  font-size: 14px;
  line-height: 1.6;
  color: #555;
}

/* 6. CTA links → primary button style (screens 1 & 4) */
#kc-content-wrapper a:not(.alert a) {
  display: inline-block;
  background: #1a5276;
  color: #fff;
  padding: 10px 24px;
  border-radius: 4px;
  text-decoration: none;
  font-weight: 500;
  font-size: 14px;
  margin-top: 12px;
  transition: background 0.15s;
}

#kc-content-wrapper a:not(.alert a):hover {
  background: #1a6da0;
}
